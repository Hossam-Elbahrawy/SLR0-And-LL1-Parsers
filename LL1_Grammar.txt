
S -> ASSIGNMENT ; | DECLARATION ; | EXPRESSION ; | epsilon
DECLARATION -> int ASSIGNMENT | str ASSIGNMENT
ASSIGNMENT -> identifier EXPRESSION_TYPE
EXPRESSION_TYPE -> NUMERICAL_EXPRESSION = | STRING_EXPRESSION = | epsilon
NUMERICAL_EXPRESSION  -> E
E -> T E_PRIME
E_PRIME -> + T E_PRIME | - T E_PRIME | epsilon
T -> F T_PRIME
T_PRIME -> * F T_PRIME | / F T_PRIME | epsilon
F -> identifier | number | - F | ( E )
STRING_EXPRESSION -> string
EXPRESSION -> IF_EXPRESSION | WHILE_EXPRESSION
IF_EXPRESSION -> if CONDITION then S else S endif
CONDITION -> ( VARIABLE COMPARE )
VARIABLE -> identifier | number | string
COMPARE -> epsilon | CONDITIONAL_OPERATOR | VARIABLE
CONDITIONAL_OPERATOR -> ?= | != | < EQ | > EQ
EQ -> = | epsilon
WHILE_EXPRESSION -> while CONDITION do S endwhile
