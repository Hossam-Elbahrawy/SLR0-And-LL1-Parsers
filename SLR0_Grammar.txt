DECLARATION-> str ASSIGNMENT
ASSIGNMENT -> identifier = EXPRESSION_TYPE

EXPRESSION_TYPE -> STRING_EXPRESSION

CONDITION -> ( VARIABLE COMPARE )

CONDITIONAL_OPERATOR -> ?=
CONDITIONAL_OPERATOR -> !=
